\chapter{ }༄༅༅།\chapter{ }།རྒྱ་གར་སྐད་དུ།\chapter{ }བོ་དྷི་ཙིཏྟ་\footnote{«སྣར་»«པེ་»ཙི་ཏཱ་}བི་བ་ར་ཎ།\chapter{ }བོད་སྐད་དུ།\chapter{ }བྱང་ཆུབ་སེམས་ཀྱི་འགྲེལ་པ།\chapter{ }འཕགས་པ་འཇམ་དཔལ་ལ་ཕྱག་འཚལ་ལོ།\chapter{ }།དེ་ནས་བྱང་ཆུབ་ཀྱི་སེམས་སྒོམ་པའི་འགྲེལ་པ་བཤད་པར་བྱའོ།\chapter{ }།བཅོམ་ལྡན་འདས་ཀྱིས་ཆོས་ཐམས་ཅད་སེམས་ཀྱིས་རྣམ་པར་བརྟགས་པའོ་ཞེས་གསུངས་པས་རེ་ཞིག་སེམས་ནི་དེ་ཉིད་ཀྱི་རང་བཞིན་ཅི་ཞིག་ཅེས་ངེས་པར་རྟོག་པ་ན།\chapter{ }དངོས་པོ་ཐམས་ཅད་དང་བྲལ་བ།\chapter{ }།ཕུང་པོ་ཁམས་དང་སྐྱེ་མཆེད་དང་།\chapter{ }།གཟུང་དང་འཛིན་པ་རྣམ་པར་སྤངས།\chapter{ }།ཆོས་བདག་མེད་པར་མཉམ་ཉིད་པས།\chapter{ }།རང་སེམས་གདོད་ནས་མ་སྐྱེས་པ།\chapter{ }།སྟོང་པ་ཉིད་ཀྱི་རང་བཞིན་ནོ།\chapter{ }།ཞེས་གསུངས་སོ།\chapter{ }།སྨྲས་པ་འདི་ཡི་དོན་ཅི་ཡིན་ཞེ་ན།\chapter{ }བརྗོད་པ།\chapter{ }:དངོས་པོ་ཐམས་ཅད་\footnote{«སྣར་»«པེ་»དངོས་པོ་}དང་བྲལ་བ།\chapter{ }།ཞེས་ཏེ་དངོས་པོའི་སྒྲས་བདག་ལ་སོགས་པ་མངོན་པར་བརྗོད་དོ།\chapter{ }།འདི་སྐད་དུ་སེམས་དེ་ནི་བདག་དང་།\chapter{ }སེམས་ཅན་དང་།\chapter{ }སྲོག་དང་།\chapter{ }གསོ་བ་དང་།\chapter{ }སྐྱེས་བུ་དང་།\chapter{ }དེ་དག་ལ་བྱ་བའི་མཚན་ཉིད་ཅན་ཡོད་པ་མ་ཡིན་པའི་ཕྱིར་རོ།\chapter{ }།དེ་ཉིད་ཀྱི་ཕྱིར་བདག་ལ་སོགས་པའི་དངོས་པོ་ཡོད་པ་མ་ཡིན་ན།\chapter{ }སེམས་དེའི་རང་བཞིན་དུ་གྱུར་པ་ལྟ་ག་ལ་ཡོད་ཅེས་སྨྲས་པར་འགྱུར་རོ།\chapter{ }།དེའི་ཕྱིར་དངོས་པོ་ཐམས་ཅད་དང་བྲལ་:ཞིང་བདག་ཏུ་\footnote{«སྣར་»«པེ་»ཞིང་བདག་དང་}འཛིན་པ་ཙམ་གྱིས་ཡང་དག་པར་བསྐྱེད་པའོ།\chapter{ }།ཉོན་མོངས་པ་ཐམས་ཅད་བདག་ཏུ་འཛིན་པས་ཡང་དག་པར་བསྐྱེད་པའི་ཕྱིར་སེམས་ནི་དེའི་རང་བཞིན་དུ་འགྱུར་རོ་ཞེ་ན།\chapter{ }སེམས་དེ་ནི་ཕུང་པོ་དང་ཁམས་དང་སྐྱེ་མཆེད་ཀྱི་རང་བཞིན་མ་ཡིན་པའི་ཕྱིར།\chapter{ }ཕུང་པོ་ཁམས་དང་སྐྱེ་མཆེད་དང་།\chapter{ }།གཟུང་དང་འཛིན་པ་རྣམ་པར་སྤངས།\chapter{ }།ཞེས་གསུངས་སོ།\chapter{ }།འདི་ལྟར་རེ་ཞིག་ཕུང་པོ་ལ་སོགས་པ་ཉིད་ཀྱི་དངོས་པོ་དོན་དམ་པར་མ་དམིགས་ན།\chapter{ }སེམས་དེའི་རང་བཞིན་དུ་འགྱུར་བ་ལྟ་ག་ལ་ཡོད།\chapter{ }ཇི་ལྟར་ཞེ་ན།\chapter{ }རེ་ཞིག་གཟུགས་ལ་སོགས་པའི་ཕུང་པོ་རྣམས་\footnote{«སྣར་»«པེ་»རྣམས་དངོས་པོ་རྣམས་}དངོས་པོ་མེད་པ་ཉིད་དེ།\chapter{ }གཟུགས་ཀྱི་ཕུང་པོ་ནི་འབྱུང་བ་དང་འབྱུང་བ་ལས་གྱུར་པའི་ཚོགས་སོ།\chapter{ }།འབྱུང་བ་ནི་བཞི་སྟེ།\chapter{ }ས་དང་།\chapter{ }ཆུ་དང་།\chapter{ }མེ་དང་།\chapter{ }རླུང་ངོ།\chapter{ }།འབྱུང་བ་ལས་གྱུར་པ་ལྔ་སྟེ།\chapter{ }གཟུགས་དང་།\chapter{ }སྒྲ་དང་།\chapter{ }དྲི་དང་།\chapter{ }རོ་དང་།\chapter{ }རེག་བྱའོ།\chapter{ }།ས་ལ་སོགས་པ་འབྱུང་བ་རྣམས་དང་།\chapter{ }གཟུགས་ལ་སོགས་པ་འབྱུང་བ་ལས་གྱུར་པ་དེ་དག་རེ་རེ་ཞིང་ཐ་དད་པའི་དངོས་པོ་ཉིད་དུ་མ་དམིགས་ཏེ།\chapter{ }གང་དུ་གཅིག་ཏུ་ཡོད་པ་དེར་གཞན་ཡང་ཡོད་པའི་ཕྱིར་རོ།\chapter{ }།དེའི་ཕྱིར་གཟུགས་ནི་མིང་ཙམ་དུ་ཀུན་དུ་བརྟགས་པའོ།\chapter{ }།གཟུགས་ཀྱི་ཕུང་པོ་ནི་གཟུགས་ཀྱི་ཕུང་པོས་སྟོང་ཞེས་གསུངས་པ་ལས་ནི་གཟུགས་ཀྱི་ཕུང་པོ་ཡང་ཡོངས་སུ་མ་གྲུབ་པའོ།\chapter{ }།དེ་བཞིན་དུ་ཚོར་བའི་ཕུང་པོ་ཡོངས་སུ་མ་གྲུབ་སྟེ།\chapter{ }ཇི་ལྟ་ཞེ་ན།\chapter{ }ཚོར་བ་:རྣམ་པ་\footnote{«སྣར་»«པེ་»རྣམ་}གསུམ་མོ།\chapter{ }།བདེ་བ་དང་།\chapter{ }སྡུག་བསྔལ་བ་དང་།\chapter{ }སྡུག་བསྔལ་མ་ཡིན་ཞིང་བདེ་བ་ཡང་མ་ཡིན་པ་དེ་དག་ཕན་ཚུན་ལྟོས་པའི་ཕྱིར་ཏེ།\chapter{ }སྡུག་བསྔལ་ལ་ལྟོས་\footnote{«སྣར་»«པེ་»བལྟོས་}ནས་བདེ་བ་དང་།\chapter{ }བདེ་བ་ལ་ལྟོས་ནས་སྡུག་བསྔལ་དུ་འདོགས་པར་བྱེད་དོ།\chapter{ }།ཆོས་གང་ཞིག་གཅིག་པུར་གནས་པ་དམིགས་པ་མེད་པ་དེ་ནི་རྫས་ཉིད་དུ་ཡོད་པ་ཡིན་ཏེ།\chapter{ }རིང་པོ་དང་ཐུང་ངུ་བཞིན་ནོ།\chapter{ }།གཞན་དུ་ཚོར་བ་ནི་རྣམ་པ་གཉིས་ཏེ།\chapter{ }ལུས་ལས་བྱུང་བ་དང་།\chapter{ }སེམས་ལས་བྱུང་བའོ་ཞེས་པར་ཡང་རེ་ཞིག་ལུས་ནི་གཟུགས་ཀྱི་ཕུང་པོས་བསྡུས་པའི་ཕྱིར་མ་དམིགས་སོ།\chapter{ }།དེའི་ཕྱིར་ལུས་མེད་པའི་ཕྱིར་ལུས་ལས་བྱུང་བའི་ཚོར་བ་མ་གྲུབ་བོ།\chapter{ }།སེམས་ཀྱང་ཡུལ་ན་མི་གནས་པ།\chapter{ }ཕྱོགས་ན་མི་གནས་པ།\chapter{ }གཟུགས་མེད་པ།\chapter{ }བསྟན་དུ་མེད་པ།\chapter{ }།དེའི་ཕྱིར་ཚོར་བའི་:ཕུང་པོ་ནི་ཚོར་བའི་ཕུང་པོས་སྟོང་ངོ་\footnote{«སྣར་»«པེ་»ཕུང་པོའོ་}ཞེས་གསུངས་པ་ལས་ན།\chapter{ }ཚོར་བའི་ཕུང་པོ་ཡང་ཡོངས་སུ་མ་གྲུབ་བོ།\chapter{ }།འདུ་ཤེས་ཀྱི་ཕུང་པོ་ཡང་མིང་ཙམ་གྱིས་དངོས་པོ་ཉིད་དེ།\chapter{ }འདུ་ཤེས་ནི་མཚན་མར་འཛིན་པའི་བདག་ཉིད་ཅན་ཡིན་ལ།\chapter{
